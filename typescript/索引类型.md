# 索引数据结构的类型

从已有的数据中读取数据结构的属性类型

```ts
const MyArray = [
  { name: 'Alice', age: 15 }, 
  { name: 'Bob', age: 23 }, 
  { name: 'Eve', age: 38 }
]
// type Person = { name: string, age: number }
type Person1 = typeof MyArray[number]

const p: Person1 = {
  name: 'xiaoqian',
  age: 11, 
  // alive: true
}

type Age = typeof MyArray[number][ 'age']
const age: Age = 11

type Age2 = Person1[ 'age' ]
const age2: Age2 = 300


```
